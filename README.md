##### Install

[Source](http://tinyurl.com/h7x9ym5)

```sh
npm install babel -g
npm install webpack -g
npm install webpack-dev-server -g
mkdir reactapp
cd reactapp/
npm init
npm install react --save
npm install react-dom --save
npm install react-core --save
npm install react-loader
npm install react-present-react
sudo npm install react-present-react
sudo npm install react-core
sudo npm install babel-preset-react
npm install babel-preset-es2015
```
